<script>
  const wordInput = document.getElementById('wordInput');
  const searchBtn = document.getElementById('searchBtn');
  const loader = document.getElementById('loader');
  const card = document.getElementById('card');
  const errorBox = document.getElementById('errorBox');
  const historySection = document.getElementById('historySection');
  const historyTags = document.getElementById('historyTags');

  let searchHistory = [];

  async function lookupWord(word) {
    word = word.trim().toLowerCase();
    if (!word) return;

    wordInput.value = word;
    card.style.display = 'none';
    errorBox.style.display = 'none';
    loader.style.display = 'block';
    searchBtn.disabled = true;

    try {
      const res = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          model: "claude-sonnet-4-20250514",
          max_tokens: 1000,
          system: `You are a dictionary. Respond ONLY with a JSON object — no markdown, no backticks, no explanation. Use this exact format:
{
  "word": "string",
  "partOfSpeech": "string",
  "definition": "string",
  "example": "string",
  "synonyms": ["word1","word2","word3","word4","word5"]
}`,
          messages: [{ role: "user", content: `Define: ${word}` }]
        })
      });

      const data = await res.json();
      const raw = data.content.map(b => b.text || '').join('');
      const clean = raw.replace(/```json|```/g, '').trim();
      const result = JSON.parse(clean);

      document.getElementById('wordDisplay').textContent = result.word;
      document.getElementById('posDisplay').textContent = result.partOfSpeech;
      document.getElementById('definitionText').textContent = result.definition;
      document.getElementById('exampleText').textContent = `"${result.example}"`;

      const synWrap = document.getElementById('synonymsWrap');
      synWrap.innerHTML = '';
      (result.synonyms || []).forEach(syn => {
        const tag = document.createElement('span');
        tag.className = 'syn-tag';
        tag.textContent = syn;
        tag.onclick = () => lookupWord(syn);
        synWrap.appendChild(tag);
      });

      loader.style.display = 'none';
      card.style.display = 'block';

      if (!searchHistory.includes(result.word)) {
        searchHistory.unshift(result.word);
        if (searchHistory.length > 8) searchHistory.pop();
        renderHistory();
      }

    } catch (err) {
      loader.style.display = 'none';
      errorBox.style.display = 'block';
      errorBox.textContent = 'Something went wrong. Please try again.';
    }

    searchBtn.disabled = false;
  }

  function renderHistory() {
    if (!searchHistory.length) { historySection.style.display = 'none'; return; }
    historySection.style.display = 'block';
    historyTags.innerHTML = '';
    searchHistory.forEach(w => {
      const tag = document.createElement('span');
      tag.className = 'history-tag';
      tag.textContent = w;
      tag.onclick = () => lookupWord(w);
      historyTags.appendChild(tag);
    });
  }

  searchBtn.addEventListener('click', () => lookupWord(wordInput.value));
  wordInput.addEventListener('keydown', e => {
    if (e.key === 'Enter') lookupWord(wordInput.value);
  });
</script>
